-- Cria tablespace p/ dados
--drop tablespace tbs_dados;

create tablespace tbs_dados
datafile '/u01/app/oracle/oradata/tbd_dados.dbf' size 10M reuse
autoextend on next 10M maxsize 100M
online;


-- Cria usuario (dono das tabelas)
create user <NOME_USUÁRIO> --usuario
identified by "<SENHA_USUARIO>"  --senha
default tablespace tbs_dados
temporary tablespace temp;

-- Cria e define a "role" de privilegios (perfil)
create role perfil_desenv;

grant
create cluster,
create database link,
create procedure,
create session,
create sequence,
create synonym,
create table,
create any type,
create trigger,
create view
to perfil_desenv;

-- APENAS PARA MOSTRAR QUE PODE APLICAR ISOLADAMENTE OU EM GRUPO
grant alter session to perfil_desenv;

-- APENAS PARA MOSTRAR QUE PODE ATRIBUIR AO NOME_USUÁRIO
-- NÃO É NECESSÁRIO APLICAR POIS TUDO ESTÁ CONTIDO NO PERFIL
grant create trigger to <NOME_USUÁRIO>;

grant perfil_desenv to <NOME_USUÁRIO>;

grant unlimited tablespace to <NOME_USUÁRIO>;
